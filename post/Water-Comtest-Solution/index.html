「T3」A-B

分两种情况。

第一种，n%2=0n\%2=0n%2=0，则有 a=2na=2na=2n，b=nb=nb=n。由 2∣n2|n2∣n，则 π(2n)=π(n)\pi(2n)=\pi(n)π(2n)=π(n).

第二种，n%2=1n\%2=1n%2=1，则找到最小的奇质数 ppp，则 a=pn,b=(p−1)na=pn,b=(p-1)na=pn,b=(p−1)n。

代码：

#include<bits/stdc++.h>
using namespace std;
int n,p;
bool primes[10000001];
int main(){
	cin>>n;
	if(n%2==0) cout<<2*n<<' '<<n;
	else{
    	for(int i=2;i<=sqrt(n);i++) for(int j=2;i*j<=n;j++) primes[i*j]=true;
		for(int i=2;i<=n;i++){
			if(!primes[i] && i%2 && n%i!=0) {
		    	p=i;
	    		break;
	    	}
		}
		cout<<p*n<<' '<<(p-1)*n;
	}
}
