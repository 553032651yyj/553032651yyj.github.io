选数问题

原命题等价于求 max⁡1≤m≤nd(m)\max\limits_{1\le m\le n}^{} d(m)1≤m≤nmax​d(m)。其中 d(m)d(m)d(m) 表示正整数 mmm 的正约数个数。从最小的质数开始枚举，可以解出最优解。可以使用深搜。此处因数据不大，可以枚举。

#include<bits/stdc++.h>
using namespace std;
int p[8]={0,2,3,5,7,11,13,17};
int dm=-1;
int n,top;
int max(int a,int b){
	return a>b?a:b;
} 
int main(){
	cin>>n;
	int ans=1;
	if(n<=3) cout<<0;
	else{
		for(int i=1;ans*p[i]<=n;i++){
		ans*=p[i];
		top=i;
 	}
	if(top<=1){
		for(int i=1;pow(2,i)<=n;i++){
	    	if(pow(2,i)==n) dm=max(dm,int(log2(n)));
	    	else dm=max(dm,int(log2(n))+1);
		}
	}	
    if(top<=2){
    	int a=1,b=1;
    	for(a=1;pow(p[1],a)*pow(p[2],b)<=n;a++){
    		for(b=1;pow(p[1],a)*pow(p[2],b)<=n;b++)
    			dm=max(dm,(a+1)*(b+1)); 
			b=1;
		}	
	} 
	if(top<=3){
    	int a=1,b=1,c=1;
    	for(a=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)<=n;a++){
    		for(b=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)<=n;b++){
    			for(c=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)<=n;c++)
    		    	dm=max(dm,(a+1)*(b+1)*(c+1)); 
    		    c=1;
			}
			b=1;
		}	
	} 
	if(top<=4){
    	int a=1,b=1,c=1,d=1;
    	for(a=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)<=n;a++){
    		for(b=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)<=n;b++){
    			for(c=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)<=n;c++){
    				for(d=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)<=n;d++)
    		        	dm=max(dm,(a+1)*(b+1)*(c+1)*(d+1)); 
    		        d=1;
				}
    		    c=1;
			}
			b=1;
		}	
	}
	if(top<=5){
    	int a=1,b=1,c=1,d=1,e=1;
    	for(a=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)<=n;a++){
    		for(b=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)<=n;b++){
    			for(c=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)<=n;c++){
    				for(d=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)<=n;d++){
    					for(e=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)<=n;e++)
    		        	    dm=max(dm,(a+1)*(b+1)*(c+1)*(d+1)*(e+1)); 
    		        	e=1;
					}
    		        d=1;
				}
    		    c=1;
			}
			b=1;
		}	
	} 
	if(top<=6){
    	int a=1,b=1,c=1,d=1,e=1,f=1;
    	for(a=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)<=n;a++){
    		for(b=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)<=n;b++){
    			for(c=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)<=n;c++){
    				for(d=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)<=n;d++){
    					for(e=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)<=n;e++){
    						for(f=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)<=n;f++)
						    	dm=max(dm,(a+1)*(b+1)*(c+1)*(d+1)*(e+1)*(f+1));
						    f=1;
						}
    		        	e=1;  
					}
    		        d=1;
				}
    		    c=1;
			}
			b=1;
		}	
	} 
	if(top<=7){
    	int a=1,b=1,c=1,d=1,e=1,f=1,g=1;
    	for(a=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;a++){
    		for(b=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;b++){
    			for(c=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;c++){
    				for(d=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;d++){
    					for(e=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;e++){
    						for(f=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;f++){
    							for(g=1;pow(p[1],a)*pow(p[2],b)*pow(p[3],c)*pow(p[4],d)*pow(p[5],e)*pow(p[6],f)*pow(p[7],g)<=n;g++)
								    dm=max(dm,(a+1)*(b+1)*(c+1)*(d+1)*(e+1)*(f+1)*(g+1));
								g=1;
							}
						    f=1;
						}
    		        	e=1;  
					}
    		        d=1;
				}
    		    c=1;
			}
			b=1;
		}	
	} 
	cout<<dm;
	}
}
