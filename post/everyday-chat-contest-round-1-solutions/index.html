「ECC T2」分数

算法标签：「数论，数学」「组合数学」

思路：

设 a,b,c,da,b,c,da,b,c,d 均已约分，那么

ba<b+da+c<dc\dfrac{b}{a}<\dfrac{b+d}{a+c}<\dfrac{d}{c}ab​<a+cb+d​<cd​

不难证明，约分 b+da+c\dfrac{b+d}{a+c}a+cb+d​ ，得到 ω\omegaω 最小值。

正确性

ba<b+da+c\dfrac{b}{a}<\dfrac{b+d}{a+c}ab​<a+cb+d​

⇔ab+bc<ab+ad\Leftrightarrow ab+bc<ab+ad⇔ab+bc<ab+ad

⇔bc<ad\Leftrightarrow bc<ad⇔bc<ad

⇔ba<dc\Leftrightarrow \dfrac{b}{a}<\dfrac{d}{c}⇔ab​<cd​

另一边同理可证。

证明

设 ba<qp<dc\dfrac{b}{a}<\dfrac{q}{p}<\dfrac{d}{c}ab​<pq​<cd​ 是一组解。

则 ba<qp\dfrac{b}{a}<\dfrac{q}{p}ab​<pq​，aq<bpaq<bpaq<bp

由整数离散性可知，aq≤bp−1aq\le bp-1aq≤bp−1

将 qqq 表示出来，即 q≤bp−1aq\le \dfrac{bp-1}{a}q≤abp−1​

同理，qp<dc\dfrac{q}{p}<\dfrac{d}{c}pq​<cd​ ，pd≤qc−1pd\le qc-1pd≤qc−1， q≥pd+1cq\ge\dfrac{pd+1}{c}q≥cpd+1​

可知 pd+1c≤q≤bp−1a\dfrac{pd+1}{c}\le q\le\dfrac{bp-1}{a}cpd+1​≤q≤abp−1​

即 pd+1c≤bp−1a\dfrac{pd+1}{c}\le\dfrac{bp-1}{a}cpd+1​≤abp−1​

bcp−c≥adp+abcp-c\ge adp+abcp−c≥adp+a

bcp−adp≥a+cbcp-adp\ge a+cbcp−adp≥a+c

p≥a+cbc−adp\ge \dfrac{a+c}{bc-ad}p≥bc−ada+c​

同理 q≥b+dbc−adq\ge \dfrac{b+d}{bc-ad}q≥bc−adb+d​

p,qp,qp,q 均取等号时， qp\dfrac{q}{p}pq​ 最小为 b+da+c\dfrac{b+d}{a+c}a+cb+d​

//代码来源：553032651yyj
#include<bits/stdc++.h>

long long gcd(long long x,long long y){
	return (y==0)?x:gcd(y,x%y);
}
long long a,b,c,d,h,i,j;
int main(){
    cin>>a>>b>>c>>d;
    long long f=gcd(a,b),g=gcd(c,d);
    a/=f,b/=f;
    c/=g,d/=g;
    j=gcd(a+c,b+d);
	h=(a+c)/j;i=(b+d)/j;
    cout<<h+i;
}

「ECC T3」代码

算法标签：「模拟」「深度优先搜索，dfs」

思路：直接模拟。

代码：

//代码来源：liqingyang
#include<iostream>
using namespace std;
int num=0,ans=0;
bool c[30],mark[30];
string str;
bool jisuan(bool a,char s,bool b)
{
	return (s=='|')?(a|b):(a&b);
}
void dfs(int now)
{
	if(now==26)
	{
		bool x;
		for(int i=0;i<str.size();)
		{
			if(str[i+1]=='(')
			{
				if(str[i+2]=='~')
				{
					x=jisuan(x,str[i],!c[str[i+3]-'A']);
					i+=5;
				}
				else
				{
					int j=i+2;
					while(str[++j]!=')');
					bool last=c[str[i+2]-'A'];
					for(int k=i+3;k<j;k+=2)
					{
						last=jisuan(last,str[k],c[str[k+1]-'A']);
					}
					x=jisuan(x,str[i],last);
					i=j+1;
				}
			}
			else if(str[i]=='(')
			{
				if(str[i+1]=='~')
				{
					x=!c[str[i+2]-'A'];
					i+=4;
				}
				else
				{
					int j=i+1;
					while(str[++j]!=')');
					x=c[str[i+1]-'A'];
					for(int k=i+2;k<j;k+=2)
					{
						x=jisuan(x,str[k],c[str[k+1]-'A']);
					}
					i=j+1;
				}
			}
			else if(str[i]>='A'&&str[i]<='Z')
			{
				x=c[str[i]-'A'];
				i++;
			}
			else
			{
				x=jisuan(x,str[i],c[str[i+1]-'A']);
				i+=2;	
			}
		}
		ans+=x;
		return;
	}
	c[now]=0;
	dfs(now+1);
	if(mark[now])
	{
		c[now]=1;
		dfs(now+1);
	}
}
int main()
{
	cin>>str;
	for(int i=0;i<str.size();i++)
	{
		if(str[i]>='A'&&str[i]<='Z')
		{
			mark[str[i]-'A']=1;
		}
	}
	dfs(0);
	cout<<ans<<endl;
	return 0;
}

「ECC T4」序列

算法标签：「哈希，Hash」

思路：优化一下空间即可。

此题可能为模板题。

时限开到700ms并没有卡掉非正解。

代码(c++)：

//代码来源:ScanfN
#include<cstdio>
const int MOD=1226959;    //此处非常毒瘤，如果模数没选好Subtask3全TLE
struct Node{
	int key;
	int val;
}p[MOD];
int find(int key){
	int tmp=key%MOD;
	while(p[tmp].key!=0&&p[tmp].key!=key){
		tmp=(tmp+MOD-1)%MOD;
	}
	if(p[tmp].key==0){
		p[tmp].key=key;
	}
	return tmp;
}
int main(){
	int t;
	scanf("%d\n",&t);
	while(t--){
		char s[10];
		int x;
		scanf("%s %d\n",s,&x);
		int w=find(x);
		switch(s[0]){
			case 'a':{
				++p[w].val;
				break;
			}
			case 'd':{
				if(p[w].val!=0){
					--p[w].val;
				}
				else{
					puts("Failed");
				}
				break;
			}
			default:{
				if(p[w].val!=0){
					puts("True");
				}
				else{
					puts("False");
				}
				break;
			}
		}
	}
	return 0;
}
